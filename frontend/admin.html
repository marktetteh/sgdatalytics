Key SQL Queries

Production-ready queries powering the SG Datalytics REST API and data warehouse.

1. Get time-series for one indicator across multiple countries
Core API query
-- Powers: GET /api/data?indicator=NY.GDP.MKTP.CD&countries=GH,NG,KE
SELECT
  TRIM(c.code) AS country_code,
  c.name, c.flag, c.region,
  i.wb_code, i.name AS indicator_name, i.unit,
  dp.year, dp.value
FROM data_points dp
JOIN countries c ON dp.country_id = c.id
JOIN indicators i ON dp.indicator_id = i.id
WHERE i.wb_code = 'NY.GDP.MKTP.CD'
  AND TRIM(c.code) IN ('GH', 'NG', 'KE')
  AND dp.year BETWEEN 2015 AND 2023
ORDER BY c.name, dp.year;
2. Latest value per country for any indicator (ranking)
Ranking query
-- Powers: GET /api/latest?indicator=FP.CPI.TOTL.ZG
SELECT TRIM(c.code), c.name, c.flag, dp.year, dp.value
FROM data_points dp
JOIN countries c ON dp.country_id = c.id
JOIN indicators i ON dp.indicator_id = i.id
JOIN (
  SELECT country_id, indicator_id, MAX(year) AS max_year
  FROM data_points GROUP BY country_id, indicator_id
) latest ON dp.country_id = latest.country_id
        AND dp.indicator_id = latest.indicator_id
        AND dp.year = latest.max_year
WHERE i.wb_code = 'FP.CPI.TOTL.ZG'
ORDER BY dp.value DESC NULLS LAST;
3. Warehouse health check — coverage by sector & country
Admin query
-- Shows how many data points exist per sector per country
SELECT
  s.icon, s.name AS sector,
  TRIM(c.code) AS country,
  COUNT(dp.id) AS data_points,
  MIN(dp.year) AS yr_from,
  MAX(dp.year) AS yr_to
FROM data_points dp
JOIN countries c ON dp.country_id = c.id
JOIN indicators i ON dp.indicator_id = i.id
JOIN sectors s ON i.sector_id = s.id
GROUP BY s.id, s.icon, s.name, TRIM(c.code)
ORDER BY s.name, COUNT(dp.id) DESC;
4. Year-on-year growth calculation
Analytics query
-- Calculates % change year-over-year using LAG window function
SELECT
  TRIM(c.code) AS country,
  dp.year,
  dp.value,
  LAG(dp.value) OVER (PARTITION BY dp.country_id ORDER BY dp.year) AS prev_value,
  ROUND(
    (dp.value - LAG(dp.value) OVER (PARTITION BY dp.country_id ORDER BY dp.year))
    / ABS(NULLIF(LAG(dp.value) OVER (PARTITION BY dp.country_id ORDER BY dp.year), 0)) * 100, 2
  ) AS yoy_pct_change
FROM data_points dp
JOIN countries c ON dp.country_id = c.id
JOIN indicators i ON dp.indicator_id = i.id
WHERE i.wb_code = 'NY.GDP.MKTP.KD.ZG'
ORDER BY c.name, dp.year;
